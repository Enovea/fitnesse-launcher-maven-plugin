88  	private static void executeSingleCommand(Arguments arguments,
89  			FitNesse fitnesse, FitNesseContext context) throws Exception {
90  		TestTextFormatter.finalErrorCount = 0;
91  		System.out.println("Executing command: " + arguments.getCommand());
92  
93  		OutputStream os;
94  
95  		boolean outputRedirectedToFile = arguments.getOutput() != null;
96  
97  		if (outputRedirectedToFile) {
98  			System.out.println("-----Command Output redirected to "
99  					+ arguments.getOutput() + "-----");
100 			os = new FileOutputStream(arguments.getOutput());
101 		} else {
102 			System.out.println("-----Command Output-----");
103 			os = System.out;
104 		}
105 
106 		fitnesse.executeSingleCommand(arguments.getCommand(), os);
107 		fitnesse.stop();
108 
109 		if (outputRedirectedToFile) {
110 			os.close();
111 		} else {
112 			System.out.println("-----Command Complete-----");
113 		}
114 
115 		if (shouldExitAfterSingleCommand()) {
116 			System.exit(TestTextFormatter.finalErrorCount);
117 		}
118 	}

124 	private static FitNesseContext loadContext(Arguments arguments)
125 			throws Exception {
126 		Builder builder = new Builder();
127 		WikiPageFactory wikiPageFactory = new WikiPageFactory();
128 		ComponentFactory componentFactory = new ComponentFactory(
129 				arguments.getRootPath());
130 
131 		builder.port = arguments.getPort();
132 		builder.rootPath = arguments.getRootPath();
133 		builder.rootDirectoryName = arguments.getRootDirectory();
134 
135 		builder.pageTheme = componentFactory
136 				.getProperty(ComponentFactory.THEME);
137 		builder.defaultNewPageContent = componentFactory
138 				.getProperty(ComponentFactory.DEFAULT_NEWPAGE_CONTENT);
139 
140 		builder.root = wikiPageFactory.makeRootPage(builder.rootPath,
141 				builder.rootDirectoryName, componentFactory);
142 
143 		builder.logger = makeLogger(arguments);
144 		builder.authenticator = makeAuthenticator(arguments.getUserpass(),
145 				componentFactory);
146 
147 		FitNesseContext context = builder.createFitNesseContext();
148 
149 		extraOutput = componentFactory.loadPlugins(context.responderFactory,
150 				wikiPageFactory);
151 		extraOutput += componentFactory.loadWikiPage(wikiPageFactory);
152 		extraOutput += componentFactory
153 				.loadResponders(context.responderFactory);
154 		extraOutput += componentFactory.loadSymbolTypes();
155 		extraOutput += componentFactory.loadContentFilter();
156 		extraOutput += componentFactory.loadSlimTables();
157 
158 		WikiImportTestEventListener.register();
159 
160 		return context;
161 	}

205 	public static Authenticator makeAuthenticator(
206 			String authenticationParameter, ComponentFactory componentFactory)
207 			throws Exception {
208 		Authenticator authenticator = new PromiscuousAuthenticator();
209 		if (authenticationParameter != null) {
210 			if (new File(authenticationParameter).exists())
211 				authenticator = new MultiUserAuthenticator(
212 						authenticationParameter);
213 			else {
214 				String[] values = authenticationParameter.split(":");
215 				authenticator = new OneUserAuthenticator(values[0], values[1]);
216 			}
217 		}
218 
219 		return componentFactory.getAuthenticator(authenticator);
220 	}

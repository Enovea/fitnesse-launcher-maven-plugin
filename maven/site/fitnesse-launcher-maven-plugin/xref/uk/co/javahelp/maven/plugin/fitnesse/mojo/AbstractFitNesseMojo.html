213     protected final String calcWikiFormatClasspath() {
214         final Set<Artifact> artifacts = new LinkedHashSet<Artifact>();
215         
216    		Map<String, Artifact> dependencyArtifactMap = this.pluginDescriptor.getArtifactMap(); 
217         // We should always have FitNesse itself on the FitNesse classpath!
218        	artifacts.addAll(resolveDependencyKey(FitNesse.artifactKey, dependencyArtifactMap));
219                 
220        	// We check plugin for null to allow use in standalone mode
221         final Plugin fitnessePlugin = this.project.getPlugin(this.pluginDescriptor.getPluginLookupKey());
222        	if(fitnessePlugin == null) {
223             getLog().info("Running standalone - launching vanilla FitNesse");
224        	} else {
225             final List<Dependency> dependecies = fitnessePlugin.getDependencies();
226             if(dependecies != null && !dependecies.isEmpty()) {
227                 getLog().info("Using dependencies specified in plugin config");
228         	    for(Dependency dependency : dependecies) {
229         			final String key = dependency.getGroupId() + ":" + dependency.getArtifactId();
230         			artifacts.addAll(resolveDependencyKey(key, dependencyArtifactMap));
231         		}
232         	}
233         }
234        	
235        	if(!this.useProjectDependencies.isEmpty()) {
236             getLog().info("Using dependencies in the following scopes: " + this.useProjectDependencies);
237        		dependencyArtifactMap = ArtifactUtils.artifactMapByVersionlessId(this.project.getDependencyArtifacts());
238         	final List<Dependency> dependecies = this.project.getDependencies();
239 			for(Dependency dependency : dependecies) {
240 		    	final String key = dependency.getGroupId() + ":" + dependency.getArtifactId();
241        	    	if(this.useProjectDependencies.contains(dependency.getScope())) {
242         			artifacts.addAll(resolveDependencyKey(key, dependencyArtifactMap));
243         		}
244        		}
245        	}
246        	
247         final StringBuilder wikiFormatClasspath = new StringBuilder("\n");
248 		final ClassRealm realm = this.pluginDescriptor.getClassRealm();
249 	    setupLocalTestClasspath(realm, wikiFormatClasspath);
250         for (Artifact artifact : artifacts) {
251 		    final File artifactFile = artifact.getFile();
252             if(artifactFile != null) {
253                 getLog().debug(String.format("Adding artifact to FitNesse classpath [%s]", artifact));
254 				this.fitNesseHelper.formatAndAppendClasspathArtifact(wikiFormatClasspath, artifact);
255     	        addToRealm(realm, artifactFile);
256             } else {
257                 getLog().warn(String.format("File for artifact [%s] is not found", artifact));
258             }
259         }
260         return wikiFormatClasspath.toString();
261     }

268 	private String handleWhitespace(final String directory) {
269 		if(Utils.whitespaceSituation(directory)) {
270             try {
271         		final String relativePath = Utils.getRelativePath(new File("."), new File(directory));
272         		if(!Utils.whitespaceSituation(relativePath)) {
273             		getLog().warn(Utils.whitespaceWarning(directory, "Attempting relative path workaround"));
274             		return relativePath;
275         		}
276 			} catch (final IOException e) {
277             	getLog().error(e);
278 			}
279 		}
280 		return directory;

302     private Set<Artifact> resolveDependencyKey(final String key, final Map<String, Artifact> artifactMap) {
303        	final Artifact artifact = artifactMap.get(key);
304        	if(artifact == null) {
305             getLog().error(String.format("Lookup for artifact [%s] failed", key));
306             return Collections.emptySet();
307        	}
308         return resolveArtifactTransitively(artifact);
309     }
310 
311     private Set<Artifact> resolveArtifactTransitively(final Artifact artifact) {
312         final ArtifactResolutionRequest request = new ArtifactResolutionRequest()
313             .setArtifact( artifact )
314 			.setResolveRoot( true )
315 			.setResolveTransitively( true )
316 			.setRemoteRepositories( this.remoteArtifactRepositories )
317 			.setLocalRepository( this.localRepository );
318 		final ArtifactResolutionResult result = this.resolver.resolve(request);
319         if(!result.isSuccess()) {
320             for(Artifact missing : result.getMissingArtifacts()) {
321     			getLog().error(String.format("Could not resolve artifact: [%s]", missing));
322         	}
323         	if(result.hasExceptions() && getLog().isDebugEnabled()) {
324             	for(Exception exception : result.getExceptions()) {
325     		    	getLog().debug(exception);
326             	}
327         	}
328         }
329    		final Set<Artifact> dependencies = result.getArtifacts();
330 		return dependencies;
331     }

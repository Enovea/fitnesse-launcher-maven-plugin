270 	private void addPluginArtifacts(final Set<Artifact> artifacts) {
271    		final Map<String, Artifact> dependencyArtifactMap = this.pluginDescriptor.getArtifactMap(); 
272         // We should always have FitNesse itself on the FitNesse classpath!
273        	artifacts.addAll(resolveDependencyKey(FitNesse.artifactKey, dependencyArtifactMap));
274                 
275        	// We check plugin for null to allow use in standalone mode
276         final Plugin fitnessePlugin = this.project.getPlugin(this.pluginDescriptor.getPluginLookupKey());
277        	if(fitnessePlugin == null) {
278             getLog().info("Running standalone - launching vanilla FitNesse");
279        	} else {
280             final List<Dependency> dependecies = fitnessePlugin.getDependencies();
281             if(dependecies != null && !dependecies.isEmpty()) {
282                 getLog().info("Using dependencies specified in plugin config");
283         	    for(Dependency dependency : dependecies) {
284         			final String key = dependency.getGroupId() + ":" + dependency.getArtifactId();
285         			artifacts.addAll(resolveDependencyKey(key, dependencyArtifactMap));
286         		}
287         	}
288         }

300 	private void addProjectArtifacts(final Set<Artifact> artifacts) {
301        	if(!this.useProjectDependencies.isEmpty()) {
302             getLog().info("Using dependencies in the following scopes: " + this.useProjectDependencies);
303             if(!this.excludeOptionalDependencies) {
304             	getLog().info("Including transitive dependencies which are optional");
305                 resolveProjectDependencies(artifacts);
306         	} else {
307 		        for(Artifact artifact : this.project.getArtifacts()) {
308    	    			if(this.useProjectDependencies.contains(artifact.getScope())) {
309        					artifacts.add(artifact);
310        				}
311    				}
312         	}
313        	}

316 	private String addArtifactsToClasspath(final Set<Artifact> artifacts) {
317         final StringBuilder wikiFormatClasspath = new StringBuilder("\n");
318 		final ClassRealm realm = this.pluginDescriptor.getClassRealm();
319 	    setupLocalTestClasspath(realm, wikiFormatClasspath);
320         for (Artifact artifact : artifacts) {
321 		    final File artifactFile = artifact.getFile();
322             if(artifactFile != null) {
323                 getLog().debug(String.format("Adding artifact to FitNesse classpath [%s]", artifact));
324 				this.fitNesseHelper.formatAndAppendClasspathArtifact(wikiFormatClasspath, artifact);
325     	        addToRealm(realm, artifactFile);
326             } else {
327                 getLog().warn(String.format("File for artifact [%s] is not found", artifact));
328             }
329         }
330         return wikiFormatClasspath.toString();
331     }

389     private Set<Artifact> resolveDependencyKey(final String key, final Map<String, Artifact> artifactMap) {
390        	final Artifact artifact = artifactMap.get(key);
391        	if(artifact == null) {
392             getLog().error(String.format("Lookup for artifact [%s] failed", key));
393             return Collections.emptySet();
394        	}
395         return resolveArtifactTransitively(artifact);
396     }

398     private Set<Artifact> resolveArtifactTransitively(final Artifact artifact) {
399         final ArtifactResolutionRequest request = new ArtifactResolutionRequest()
400             .setArtifact( artifact )
401 			.setResolveRoot( true )
402 			.setResolveTransitively( true )
403 			.setRemoteRepositories( this.remoteArtifactRepositories )
404 			.setLocalRepository( this.localRepository );
405         if(this.excludeOptionalDependencies) {
406 			request.setCollectionFilter(OptionalArtifactFilter.INSTANCE);
407         }
408 		final ArtifactResolutionResult result = this.resolver.resolve(request);
409         if(!result.isSuccess()) {
410             for(Artifact missing : result.getMissingArtifacts()) {
411     			getLog().error(String.format("Could not resolve artifact: [%s]", missing));
412         	}
413         	if(result.hasExceptions() && getLog().isDebugEnabled()) {
414             	for(Exception exception : result.getExceptions()) {
415     		    	getLog().debug(exception);
416             	}
417         	}
418         }
419    		final Set<Artifact> dependencies = result.getArtifacts();
420 		return dependencies;

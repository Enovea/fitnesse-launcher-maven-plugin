64      /**
65       * Note: Through experiment I've found that we can safely send duplicate 'create SymLink' requests - FitNesse isn't bothered
66       * @throws IOException 
67       * @see <a href="http://fitnesse.org/FitNesse.UserGuide.SymbolicLinks">FitNesse SymLink User Guide</a>
68       */
69      public int createSymLink(final String suite, final String test,
70      		final File basedir, final String testResourceDirectory, final int port) throws IOException {
71          final String linkName = calcLinkName(suite, test);
72          final String linkPath = calcLinkPath(linkName, basedir, testResourceDirectory);
73  
74          HttpURLConnection connection = null;
75          try {
76              final String urlPath = 
77                  String.format("/root?responder=symlink&linkName=%s&linkPath=%s&submit=%s",
78                  URLEncoder.encode(linkName, UTF8), URLEncoder.encode(linkPath, UTF8), URLEncoder.encode("Create/Replace", UTF8));
79              final URL url = new URL("http", "localhost", port, urlPath);
80              this.log.info("Calling " + url);
81              connection = (HttpURLConnection) url.openConnection();
82              connection.setRequestMethod("GET");
83              connection.connect();
84              final int responseCode = connection.getResponseCode();
85              this.log.info("Response code: " + responseCode);
86              return responseCode;
87          } finally {
88              if(connection != null) {
89                  connection.disconnect();
90              }
91          }
92      }

94      public String[] calcPageNameAndType(final String suite, final String test) {
95          final boolean haveSuite = !isBlank(suite);
96          final boolean haveTest = !isBlank(test);
97          if (!haveSuite && !haveTest) {
98              throw new IllegalArgumentException("No suite or test page specified");
99          } else if (haveSuite && haveTest) {
100             throw new IllegalArgumentException("Suite and test page parameters are mutually exclusive");
101         }
102 
103         final String pageName = (haveSuite) ? suite : test;
104         final String pageType = (haveSuite) ? TestHelper.PAGE_TYPE_SUITE : TestHelper.PAGE_TYPE_TEST;
105 
106         return new String[] { pageName, pageType };
107     }

109     private String calcLinkName(final String suite, final String test) {
110         final String[] pageNameAndType = calcPageNameAndType(suite, test);
111         return pageNameAndType[0];
112     }

114     private String calcLinkPath(final String linkName, final File basedir, final String testResourceDirectory) {
115         final StringBuilder linkPath = new StringBuilder(
116             basedir.toURI().toString()
117                 .replaceFirst("/[A-Z]:", "")
118 				.replaceFirst(":", "://"));
119 		linkPath.append(testResourceDirectory);
120 		linkPath.append("/");
121 		linkPath.append(linkName);
122 		return linkPath.toString();
123     }

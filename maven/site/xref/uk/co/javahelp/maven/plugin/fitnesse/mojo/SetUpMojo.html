120 		executeMojo(

126 		            	element("groupId", artifact.getGroupId()),
127 		            	element("artifactId", artifact.getArtifactId()),
128 		            	element("version", artifact.getVersion()),
129 		            	element("type", "jar"),
130 		            	element("overWrite", "false"),
131 		            	element("outputDirectory", this.workingDir),

163         final Xpp3Dom config = configuration(

166         final Xpp3Dom move = config.getChild("target").getChild("move");
167         move.setAttribute("file", "${fitnesse.working}/Resources/FitNesseRoot");
168 		move.setAttribute("todir", "${fitnesse.working}");
169 		move.setAttribute("failonerror", "false");
170 		executeMojo(
